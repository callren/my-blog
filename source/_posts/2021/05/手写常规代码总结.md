## 手写call

```JavaScript
// func.call(this, a, b)
Object.prototype._call = function(context, ...params){
    if(typeof this !== 'function'){
        throw TypeError('type is fail')
    }
    let that = context || window
    let symbol = Symbol('func')
    that[symbol] = this
    let result = that[symbol](...params)
    delete that[symbol]
    return result
}
```

## 手写bind

```JavaScript

Object.prototype._bind = function(context, ...params1){
    if(typeof this !== 'function'){
        throw TypeError('type is fail')
    }
    let that = context || window
    return (...params2) => {
        this.call(that, ...params1,)
```

## 手写instanceof

```JavaScript
function instant(L, R){
    let l = L._proto_
    let r = R.prototype
    while(true){
        if(l === null) return false
        if(l === r) return true
        l = l._proto_
    }
}
```
